def binary_cross_entropy(y_pred, y_true):
    return -(y_true * numpy.log(y_pred) + (1 - y_true) * numpy.log(1 - y_pred))

def deriv_binary_cross_entry(y_pred, y_true):
    return y_pred - y_true
